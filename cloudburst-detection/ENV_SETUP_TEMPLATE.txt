# Get these values from:
# Firebase Console → Your Project → ⚙️ Project Settings → General → Your apps → SDK setup

# 1. API Key (starts with "AIzaSy...")
NEXT_PUBLIC_FIREBASE_API_KEY=your-api-key-here

# 2. Auth Domain (ends with ".firebaseapp.com")
NEXT_PUBLIC_FIREBASE_AUTH_DOMAIN=your-project-id.firebaseapp.com

# 3. Database URL (MUST start with "https://" and end with "-default-rtdb.firebaseio.com")
NEXT_PUBLIC_FIREBASE_DATABASE_URL=https://your-project-id-default-rtdb.firebaseio.com

# 4. Project ID (your Firebase project ID)
NEXT_PUBLIC_FIREBASE_PROJECT_ID=your-project-id

# 5. Storage Bucket (ends with ".appspot.com")
NEXT_PUBLIC_FIREBASE_STORAGE_BUCKET=your-project-id.appspot.com

# 6. Messaging Sender ID (12-digit number)
NEXT_PUBLIC_FIREBASE_MESSAGING_SENDER_ID=123456789012

# 7. App ID (starts with "1:" and contains "web:")
NEXT_PUBLIC_FIREBASE_APP_ID=1:123456789012:web:abcdef1234567890
